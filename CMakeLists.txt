file(GLOB _SRCS CONFIGURE_DEPENDS
  src/*.cpp
  src/*.hpp
  src/*.h
)
target_sources(${LIB_COINUTILS} PRIVATE ${_SRCS})

set(COINUTILS_COMPILE_DEFINITIONS ${COMMON_COMPILE_DEFINITIONS})
list(APPEND COINUTILS_COMPILE_DEFINITIONS
  "COINUTILSLIB_BUILD"
  "_LIB"
)

target_compile_definitions(${LIB_COINUTILS} PRIVATE ${COINUTILS_COMPILE_DEFINITIONS})
target_compile_options(${LIB_COINUTILS} PRIVATE ${COMMON_COMPILE_OPTIONS})
target_link_options(${LIB_COINUTILS} PRIVATE ${COMMON_LINK_OPTIONS})
target_include_directories(${LIB_COINUTILS} PUBLIC
  ${CMAKE_CURRENT_SOURCE_DIR}/src
)
